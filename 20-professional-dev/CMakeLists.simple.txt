# Simple CMake Example
# Demonstrates basic CMake usage

# Minimum CMake version required
cmake_minimum_required(VERSION 3.10)

# Project name and version
project(SimpleProject VERSION 1.0 LANGUAGES C)

# C standard
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Source files
set(SOURCES
    main.c
    utils.c
)

# Create executable
add_executable(simple_program ${SOURCES})

# Compiler warnings
target_compile_options(simple_program PRIVATE
    -Wall
    -Wextra
    -Wpedantic
)

# Optional: Link math library
target_link_libraries(simple_program PRIVATE m)

# Print configuration
message(STATUS "Building ${PROJECT_NAME} version ${PROJECT_VERSION}")
message(STATUS "C Standard: ${CMAKE_C_STANDARD}")

# Installation
install(TARGETS simple_program
    RUNTIME DESTINATION bin
)
